258void World::initializeTiles()
259{
260 tiles_.clear();
261 tiles_.resize(static_cast<size_t>(mapHeight_));
262 for (int y = 0; y < mapHeight_; ++y) {
263 tiles_[static_cast<size_t>(y)].clear();
264 for (int x = 0; x < mapWidth_; ++x) {
265 tiles_[static_cast<size_t>(y)].push_back(Tile(x, y));
266 }
267 }
268 LOG_DEBUGF("Initialized %dx%d tile grid", mapWidth_, mapHeight_);
269}
