10static int create_socket_with_options(void)
11{
12 int server_fd;
13 int opt = 1;
14
15 server_fd = socket(AF_INET, SOCK_STREAM, 0);
16 if (server_fd == -1) {
17 perror("socket");
18 return -1;
19 }
20 if (setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR,
21 &opt, sizeof(opt)) == -1) {
22 perror("setsockopt");
23 close(server_fd);
24 return -1;
25 }

29static int bind_and_listen(int server_fd, int port)
30{
31 struct sockaddr_in server_addr;
32
33 memset(&server_addr, 0, sizeof(server_addr));
34 server_addr.sin_family = AF_INET;
35 server_addr.sin_addr.s_addr = INADDR_ANY;
36 server_addr.sin_port = htons(port);
37 if (bind(server_fd, (struct sockaddr *)&server_addr,
38 sizeof(server_addr)) == -1) {
39 perror("bind");
40 close(server_fd);
41 return -1;
42 }
43 if (listen(server_fd, SOMAXCONN) == -1) {
44 perror("listen");
45 close(server_fd);
46 return -1;
47 }
48 return 0;
