239void Camera::resetView()
240{
241 switch (mode_) {
242 case CameraMode::FREE:
243 break;
244 case CameraMode::FOLLOW:
245 break;
246 case CameraMode::TEAM:
247 break;
248 case CameraMode::OVERVIEW:
249 float mapWidth = maxBounds_.x - minBounds_.x;
250 float mapHeight = maxBounds_.y - minBounds_.y;
251 if (mapWidth > 0 && mapHeight > 0) {
252 float zoomX = viewSize_.x / mapWidth;
253 float zoomY = viewSize_.y / mapHeight;
254 float optimalZoom = std::min(zoomX, zoomY) * 1.0f;
255 setZoom(std::max(0.1f, std::min(2.0f, optimalZoom)));
256 setPosition((minBounds_.x + maxBounds_.x) / 2, (minBounds_.y + maxBounds_.y) / 2);
257 } else {
258 setZoom(0.5f);
259 setPosition(320.0f, 320.0f);
260 }
261 break;
262 }
263}

271{
272 sf::Vector2f halfView = viewSize_ * zoom_ * 0.5f;
273 float minX = minBounds_.x + halfView.x;
274 float maxX = maxBounds_.x - halfView.x;

280 targetPosition_.x = std::clamp(targetPosition_.x, minX, maxX);
281 }
282 if (maxY < minY) {
283 targetPosition_.y = (minBounds_.y + maxBounds_.y) / 2.0f;
284 } else {

289void Camera::smoothMovement(float deltaTime)
290{
291 if (mode_ == CameraMode::FREE) {
292 sf::Vector2f movement(0, 0);
293 float speed = moveSpeed_ * deltaTime * zoom_;
294 if (keys_[sf::Keyboard::W] || keys_[sf::Keyboard::Up]) {
295 movement.y -= speed;
296 }
297 if (keys_[sf::Keyboard::S] || keys_[sf::Keyboard::Down]) {
298 movement.y += speed;
299 }
300 if (keys_[sf::Keyboard::A] || keys_[sf::Keyboard::Left]) {
301 movement.x -= speed;
302 }
303 if (keys_[sf::Keyboard::D] || keys_[sf::Keyboard::Right]) {
304 movement.x += speed;
305 }
306 if (movement.x != 0 || movement.y != 0) {
307 move(movement.x, movement.y);
308 }
309 }
310 if (math::length(targetPosition_ - position_) > 0.1f) {
311 sf::Vector2f delta = targetPosition_ - position_;
312 position_ += delta * smoothness_ * deltaTime;
313 } else {
314 position_ = targetPosition_;
315 }
316 if (std::abs(targetZoom_ - zoom_) > 0.01f) {
317 float delta = targetZoom_ - zoom_;
318 zoom_ += delta * smoothness_ * deltaTime;
319 } else {
320 zoom_ = targetZoom_;
321 }
322}
