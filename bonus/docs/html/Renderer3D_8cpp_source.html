304void Renderer3D::handlePlayerSelection(const World& world) {
305 if (IsMouseButtonPressed(MOUSE_BUTTON_LEFT)) {
306 Ray ray = GetMouseRay(GetMousePosition(), camera3D_);
307 float closestCollision = -1.0f;
308 std::optional<int> newSelectedPlayerId;
309
310 for (const auto& pair : world.getPlayers()) {
311 const Player* player = pair.second.get();
312 if (!player) continue;
313
314 sf::Vector2f pos2D = player->getInterpolatedPosition();
315 Vector3 position = toVector3(pos2D.x / 32.0f, 1.0f, pos2D.y / 32.0f);
316 BoundingBox playerBox = {
317 (Vector3){ position.x - 0.5f, position.y - 1.0f, position.z - 0.5f },
318 (Vector3){ position.x + 0.5f, position.y + 1.0f, position.z + 0.5f }
319 };
320
321 RayCollision collision = GetRayCollisionBox(ray, playerBox);
322 if (collision.hit) {
323 if (closestCollision < 0 || collision.distance < closestCollision) {
324 closestCollision = collision.distance;
325 newSelectedPlayerId = player->getId();
326 }
327 }
328 }
329
330 if (newSelectedPlayerId) {
331 selectedPlayerId_ = newSelectedPlayerId;
332 selectedTile_ = std::nullopt;
333 }
334 }

337void Renderer3D::handleTileSelection(const World& world) {
338 if (IsMouseButtonPressed(MOUSE_BUTTON_LEFT)) {
339 if (selectedTile_) {
340 int panelWidth = 300;
341 int panelHeight = 180;
342 int panelX = GetScreenWidth() / 2 - panelWidth / 2;
343 int panelY = GetScreenHeight() - 220;
344 Rectangle panelRect = { (float)panelX, (float)panelY, (float)panelWidth, (float)panelHeight };
345
346 if (CheckCollisionPointRec(GetMousePosition(), panelRect)) {
347 return;

350
351 Ray ray = GetMouseRay(GetMousePosition(), camera3D_);
352 sf::Vector2i mapSize = world.getMapSize();
353 float closestCollision = -1.0f;
354 std::optional<sf::Vector2i> newSelectedTile;
355

358 Vector3 p1 = toVector3(x - 0.5f, 0, y - 0.5f);

361 Vector3 p4 = toVector3(x - 0.5f, 0, y + 0.5f);
362
363 RayCollision collision = GetRayCollisionQuad(ray, p1, p2, p3, p4);
364
365 if (collision.hit) {
366 if (closestCollision < 0 || collision.distance < closestCollision) {
367 closestCollision = collision.distance;
368 newSelectedTile = sf::Vector2i(x, y);

372 }
373 if (newSelectedTile) {

375 selectedPlayerId_ = std::nullopt;

380bool Renderer3D::handleMouseEvents() {
381 float wheel = GetMouseWheelMove();
382 if (wheel != 0) {
383 Vector3 direction = Vector3Subtract(camera3D_.target, camera3D_.position);
384 float distance = Vector3Length(direction);
385 direction = Vector3Normalize(direction);
386
387 distance += wheel * 2.0f;
388 distance = std::max(2.0f, std::min(distance, 50.0f));
389
390 camera3D_.position = Vector3Subtract(camera3D_.target, Vector3Scale(direction, distance));
391 return true;
392 }
393
394 if (IsMouseButtonDown(MOUSE_BUTTON_RIGHT)) {
395 Vector2 mouseDelta = GetMouseDelta();
396 if (mouseDelta.x != 0 || mouseDelta.y != 0) {
397 float sensitivity = 0.05f;
398
399 Vector3 forward = Vector3Normalize(Vector3Subtract(camera3D_.target, camera3D_.position));
400 Vector3 right = Vector3Normalize(Vector3CrossProduct(forward, camera3D_.up));
401 Vector3 up = Vector3CrossProduct(right, forward);
402
403 Vector3 panOffset = Vector3Add(
404 Vector3Scale(right, -mouseDelta.x * sensitivity),
405 Vector3Scale(up, mouseDelta.y * sensitivity)
406 );
407
408 camera3D_.position = Vector3Add(camera3D_.position, panOffset);
409 camera3D_.target = Vector3Add(camera3D_.target, panOffset);
410
411 return true;
412 }
413 }
414 return false;
415}

447void Renderer3D::renderResources(const World& world) {
448 sf::Vector2i mapSize = world.getMapSize();
449
450 for (int y = 0; y < mapSize.y; ++y) {
451 for (int x = 0; x < mapSize.x; ++x) {
452 const Tile& tile = world.getTile(x, y);
453 const auto& resources = tile.getResources();
454
455 int resourceIndex = 0;
456 for (int i = 0; i < static_cast<int>(ResourceType::Count); ++i) {
457 if (resources[i] > 0) {
458 ResourceType type = static_cast<ResourceType>(i);
459
460 float offsetX = (resourceIndex % 3 - 1) * 0.3f;
461 float offsetZ = (resourceIndex / 3 - 1) * 0.3f;
462
463 Vector3 position = toVector3(
464 static_cast<float>(x) + offsetX,
465 0.01f,
466 static_cast<float>(y) + offsetZ
467 );
468
469 if (shadowsEnabled_) {

472
