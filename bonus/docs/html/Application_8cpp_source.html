141void Application::handleEvents()
142{
143 if (config_.renderMode != Config::RenderMode::Mode2D) {
144 return;
145 }
146 sf::Event event;

402void Application::processMessage(const Message& message)
403{
404 LOG_DEBUGF("Received message: %s", message.toString().c_str());
405 switch (message.type) {
406 case MessageType::MapSize: {
407 if (message.hasMinParams(2)) {
408 auto width = message.getIntParam(0);
409 auto height = message.getIntParam(1);
410 if (width && height) {
411 world_->setMapSize(*width, *height);
412 if (camera_) {
413 camera_->setBounds(0.0f, 0.0f, *width * 32.0f, *height * 32.0f);
414 camera_->resetView();
415 LOG_DEBUGF("Camera bounds set and view reset for map %dx%d (%.1f x %.1f pixels)",
416 *width, *height, *width * 32.0f, *height * 32.0f);
417 }
418 }
419 }
420 break;
421 }
422 case MessageType::TileContent: {
423 if (message.hasMinParams(9)) {
424 auto x = message.getIntParam(0);
425 auto y = message.getIntParam(1);
426 if (x && y && world_->isValidPosition(*x, *y)) {
427 Tile& tile = world_->getTile(*x, *y);
428 std::array<int, 7> resources;
429 for (int i = 0; i < 7; ++i) {
430 auto quantity = message.getIntParam(2 + i);
431 resources[i] = quantity ? *quantity : 0;
432 }
433 tile.setResources(resources);
434 }
435 }
436 break;
437 }
438 case MessageType::TeamName: {
439 if (message.hasMinParams(1)) {
440 std::string teamName = message.getParam<std::string>(0).value();
441 TeamID newTeamId = static_cast<TeamID>(world_->getTeams().size() + 1);
442 if (!world_->getTeamByName(teamName)) {
443 world_->addTeam(std::make_unique<Team>(newTeamId, teamName));
444 LOG_INFOF("Added team: %s", teamName.c_str());
445 }
446 }
447 break;
448 }
449 case MessageType::PlayerNew: {
450 if (message.hasMinParams(6)) {
451 std::string playerIdStr = message.getParam<std::string>(0).value();
452 auto x = message.getIntParam(1);
453 auto y = message.getIntParam(2);
454 auto orientation = message.getIntParam(3);
455 auto level = message.getIntParam(4);
456 std::string teamName = message.getParam<std::string>(5).value();
457 int playerId = -1;
458 if (playerIdStr.length() > 1 && playerIdStr[0] == '#') {
459 try {
460 playerId = std::stoi(playerIdStr.substr(1));
461 } catch (const std::exception&) {
462 LOG_ERROR("Failed to parse player ID: " + playerIdStr);
463 break;
464 }
465 }
466 if (playerId >= 0 && x && y && orientation && level) {
467 Team* team = world_->getTeamByName(teamName);
468 if (!team) {
469 TeamID newTeamId = static_cast<TeamID>(world_->getTeams().size() + 1);
470 world_->addTeam(std::make_unique<Team>(newTeamId, teamName));
471 team = world_->getTeamByName(teamName);
472 }
473 auto player = std::make_unique<Player>(playerId, teamName, *x, *y,
474 static_cast<PlayerOrientation>(*orientation));
475 player->setLevel(*level);
476 if (team) {
477 player->setTeamId(team->getId());
478 }
479 world_->addPlayer(std::move(player));
480 LOG_INFOF("Added player %d at (%d, %d) for team %s", playerId, *x, *y, teamName.c_str());
481 }
482 }
483 break;
484 }
485 case MessageType::PlayerPosition: {
486 if (message.hasMinParams(4)) {
487 std::string playerIdStr = message.getParam<std::string>(0).value();
488 auto x = message.getIntParam(1);
489 auto y = message.getIntParam(2);
490 auto orientation = message.getIntParam(3);
491 int playerId = -1;
492 if (playerIdStr.length() > 1 && playerIdStr[0] == '#') {
493 try {
494 playerId = std::stoi(playerIdStr.substr(1));
495 } catch (const std::exception&) {
496 break;
497 }
498 }
499 if (playerId >= 0 && x && y && orientation) {
500 Player* player = world_->getPlayer(playerId);
501 if (player) {
502 player->setServerPosition(*x, *y);
503 player->setOrientation(static_cast<PlayerOrientation>(*orientation));
504 } else {
505 LOG_WARNINGF("Received PPO for unknown player %d", playerId);
506 }
507 }
508 }
509 break;
510 }
511 case MessageType::PlayerLevel: {
512 if (message.hasMinParams(2)) {
513 std::string playerIdStr = message.getParam<std::string>(0).value();
514 auto level = message.getIntParam(1);
515 int playerId = -1;
516 if (playerIdStr.length() > 1 && playerIdStr[0] == '#') {
517 try {
518 playerId = std::stoi(playerIdStr.substr(1));
519 } catch (const std::exception&) {
520 break;
521 }
522 }
523 if (playerId >= 0 && level) {
524 Player* player = world_->getPlayer(playerId);
525 if (player) {
526 player->setLevel(*level);
527 }
528 }
529 }
530 break;
531 }
532 case MessageType::PlayerInventory: {
533 if (message.hasMinParams(10)) {
534 std::string playerIdStr = message.getParam<std::string>(0).value();
535 auto x = message.getIntParam(1);
536 auto y = message.getIntParam(2);
537 int playerId = -1;
538 if (playerIdStr.length() > 1 && playerIdStr[0] == '#') {
539 try {
540 playerId = std::stoi(playerIdStr.substr(1));
541 } catch (const std::exception&) {
542 break;
543 }
544 }
545 if (playerId >= 0 && x && y) {
546 Player* player = world_->getPlayer(playerId);
547 if (player) {
548 player->setServerPosition(*x, *y);
549 std::array<int, 7> inventory;
550 for (int i = 0; i < 7; ++i) {
551 auto quantity = message.getIntParam(3 + i);
552 inventory[i] = quantity ? *quantity : 0;

561 if (message.hasMinParams(1)) {
562 std::string playerIdStr = message.getParam<std::string>(0).value();
563 int playerId = -1;
564 if (playerIdStr.length() > 1 && playerIdStr[0] == '#') {

633

635 if (renderer_) {
636 renderer_->triggerElevationAnimation(tilePos, player->getLevel(), player->getTeamColor());
637 }
638 LOG_INFOF("Elevation animation triggered at (%d, %d) for player %d level %d",

641 }
642 }
643 }
644 }
645 }
646 }
647 break;
648 }
