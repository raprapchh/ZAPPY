38bool ResourceManager::loadTexture(const std::string& name, const std::string& filename)
39{
40 if (hasTexture(name)) {
41 LOG_DEBUGF("Texture '%s' already loaded", name.c_str());

46 LOG_ERRORF("Failed to load texture '%s' from file '%s'", name.c_str(), filename.c_str());

48 }

180void ResourceManager::loadDefaultResources()
181{
182 LOG_DEBUG("Loading default resources...");
183 sf::Font defaultFont;
184 if (defaultFont.loadFromFile("Share/OpenSans Regular.ttf")) {
185 fonts_["default"] = std::move(defaultFont);
186 LOG_DEBUGF("Loaded default font from 'Share/OpenSans Regular.ttf'", "");
187 return;
188 } else {
189 LOG_WARNING("Could not load project font 'Share/OpenSans Regular.ttf'. Trying system fonts.");
190 }
191 std::vector<std::string> fontPaths = {
192 "/usr/share/fonts/liberation-sans/LiberationSans-Regular.ttf",
193 "/usr/share/fonts/liberation-mono/LiberationMono-Regular.ttf",
194 "/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf",
195 "/usr/share/fonts/TTF/DejaVuSans.ttf",
196 "/usr/share/fonts/truetype/liberation/LiberationSans-Regular.ttf",
197 "/System/Library/Fonts/Arial.ttf",
198 "C:/Windows/Fonts/arial.ttf"
199 };
200 bool fontLoaded = false;
201 for (const auto& path : fontPaths) {
202 if (defaultFont.loadFromFile(path)) {
203 fonts_["default"] = std::move(defaultFont);
204 LOG_DEBUGF("Loaded default font from '%s'", path.c_str());
205 fontLoaded = true;
206 break;
207 }
208 }
209 if (!fontLoaded) {
210 LOG_WARNING("Could not load any system font for default - text rendering may not work.");
211 }
212}

249void ResourceManager::createAtlasTexture()
250{
251 LOG_DEBUG("Creating atlas texture...");
252 sf::Image atlasImage;
253 atlasImage.create(256, 256, sf::Color::Transparent);
254 for (unsigned int y = 0; y < 256; ++y) {
255 for (unsigned int x = 0; x < 256; ++x) {
256 if ((x / 32 + y / 32) % 2 == 0) {
257 atlasImage.setPixel(x, y, sf::Color(100, 100, 100));
258 }
259 }
260 }
261 sf::Texture atlasTexture;
262 if (atlasTexture.loadFromImage(atlasImage)) {
263 textures_["atlas"] = std::move(atlasTexture);
264 LOG_DEBUG("Created atlas texture");
265 } else {
266 LOG_ERROR("Failed to create atlas texture");
