10resource_type_t parse_resource(const char *name)
11{
12 if (!name)
13 return -1;
14 if (strcmp(name, "food") == 0)
15 return FOOD;
16 if (strcmp(name, "linemate") == 0)
17 return LINEMATE;
18 if (strcmp(name, "deraumere") == 0)
19 return DERAUMERE;
20 if (strcmp(name, "sibur") == 0)
21 return SIBUR;
22 if (strcmp(name, "mendiane") == 0)
23 return MENDIANE;
24 if (strcmp(name, "phiras") == 0)
25 return PHIRAS;
26 if (strcmp(name, "thystame") == 0)
27 return THYSTAME;
28 return -1;
29}

31void cmd_take(server_t *server, client_t *client, const char *object)
32{
33 uint64_t execute_time = get_timestamp() + (7000 / server->config.freq);
34 char command[CMD_MAX];
