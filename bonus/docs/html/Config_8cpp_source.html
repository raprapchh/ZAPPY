163 return false;

175 std::string value = line.substr(pos + 1);
176 key.erase(0, key.find_first_not_of(" \t"));
177 key.erase(key.find_last_not_of(" \t") + 1);
178 value.erase(0, value.find_first_not_of(" \t"));
179 value.erase(value.find_last_not_of(" \t") + 1);
180 if (key == "server_host") {
181 serverHost = value;
182 } else if (key == "server_port") {
183 try {
184 serverPort = std::stoi(value);

186 std::cerr << "Warning: Invalid server_port in config: " << value << std::endl;

188 } else if (key == "window_width") {

190 windowWidth = std::stoi(value);

192 std::cerr << "Warning: Invalid window_width in config: " << value << std::endl;

194 } else if (key == "window_height") {

196 windowHeight = std::stoi(value);

198 std::cerr << "Warning: Invalid window_height in config: " << value << std::endl;
199 }
200 } else if (key == "fullscreen") {
201 fullscreen = (value == "true" || value == "1");
202 } else if (key == "vsync") {
203 vsync = (value == "true" || value == "1");
204 } else if (key == "fps_limit") {

237{
238 std::regex resolutionRegex(R"((\d+)x(\d+))");
239 std::smatch match;
240 if (std::regex_match(str, match, resolutionRegex)) {
241 try {
242 width = std::stoi(match[1].str());
