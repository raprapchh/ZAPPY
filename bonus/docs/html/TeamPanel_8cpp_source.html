41void TeamPanel::update(float deltaTime, const World& world)
42{
43 (void)deltaTime;
44 if (!visible_) return;
45 teams_.clear();
46 for (const auto& [teamId, team] : world.getTeams()) {
47 TeamDisplayInfo info;
48 info.name = team->getName();
49 info.color = team->getColor();
50 info.playerCount = 0;
51 info.eggCount = 0;
52 info.victoryProgress = 0.0f;
53 info.isWinning = false;
54 std::fill(info.levelCounts.begin(), info.levelCounts.end(), 0);
55 std::fill(info.resources.begin(), info.resources.end(), 0);
56 for (const auto& [playerId, player] : world.getPlayers()) {
57 if (player->getTeamId() == teamId) {
58 info.playerCount++;
59 int level = player->getLevel();
60 if (level >= 1 && level <= 8) {
61 info.levelCounts[level - 1]++;
62 }
63 }
64 }
65 for (const auto& [eggId, egg] : world.getEggs()) {
66 if (egg->getTeamId() == teamId) {
67 info.eggCount++;
68 }
69 }
70 info.victoryProgress = static_cast<float>(info.levelCounts[7]) / 6.0f;
71 info.isWinning = info.victoryProgress >= 1.0f;
72 teams_.push_back(info);
73 }
