66void DebugPanel::update(float deltaTime, const World& world, const Camera& camera,
67 const NetworkClient& network)
68{
69 (void)deltaTime;
70 (void)world;
71 (void)camera;
72 (void)network;
73 if (!visible_) return;
74 calculateMetrics(world, camera, network);
75 fpsHistory_[fpsHistoryIndex_] = metrics_.currentFPS;
76 fpsHistoryIndex_ = (fpsHistoryIndex_ + 1) % maxFPSHistory_;
77 float graphWidth = graphBackground_.getSize().x;
78 float graphHeight = graphBackground_.getSize().y;
79 sf::Vector2f graphPos = graphBackground_.getPosition();
80 for (size_t i = 0; i < maxFPSHistory_; ++i) {
81 size_t index = (fpsHistoryIndex_ + i) % maxFPSHistory_;
82 float fps = fpsHistory_[index];
83 float x = graphPos.x + (static_cast<float>(i) / static_cast<float>(maxFPSHistory_ - 1)) * graphWidth;
84 float y = graphPos.y + graphHeight - (fps / 120.0f) * graphHeight;
85 fpsGraph_[i].position = sf::Vector2f(x, y);
86 fpsGraph_[i].color = graphColor_;
87 }
88}
