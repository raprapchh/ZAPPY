50bool server_init(server_t *server, config_t *config)
51{
52 if (!init_server_basics(server, config))
53 return false;
54 if (!init_server_socket(server, config))
55 return false;
56 server->map = map_create(config->width, config->height);
57 if (!server->map) {
58 socket_close(server->listen_fd);
59 return false;
60 }
61 if (!init_server_teams(server, config))
62 return false;
63 server->events = event_queue_create();
64 if (!server->events) {
65 server_cleanup(server);
66 return false;
67 }
68 map_spawn_resources(server->map, config->freq);
69 server->map->last_spawn = get_timestamp();
70 server->state = GAME_RUNNING;
71 return true;
72}

74void server_cleanup(server_t *server)
75{
76 client_t *client = server->clients;
77 team_t *team = server->teams;
78 void *next;
79
80 while (client) {
81 next = client->next;
82 client_destroy(client);
83 client = next;
84 }
85 while (team) {
86 next = team->next;
87 team_destroy(team);
88 team = next;
89 }
90 if (server->map)
91 map_destroy(server->map);
92 if (server->events)
93 event_queue_destroy(server->events);
94 if (server->listen_fd != -1)
95 socket_close(server->listen_fd);
