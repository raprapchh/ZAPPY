148void SpriteManager::addSprite(const sf::Sprite& sprite, sf::Texture* texture)
149{
150 if (!batchingActive_)
151 return;
152 if (!texture) {
153 texture = const_cast<sf::Texture*>(sprite.getTexture());
154 }
155 if (!texture)
156 return;
157 auto it = textureToIndex_.find(texture);
158 size_t batchIndex;
159 if (it != textureToIndex_.end()) {
160 batchIndex = it->second;
161 } else {
162 batchIndex = batches_.size();
163 textureToIndex_[texture] = batchIndex;
164 SpriteBatch batch;
165 batch.texture = texture;
166 batch.states.texture = texture;
167 batches_.push_back(batch);
168 }
169 batches_[batchIndex].sprites.push_back(sprite);
170}

172void SpriteManager::endBatch(sf::RenderTarget& target)
