116void EventLog::renderEvents(sf::RenderTarget& target)
117{
118 float y = area_.top + padding_ + 20.0f;
119 float visibleHeight = area_.height - 2 * padding_ - 20.0f;
120 size_t startIndex = scrollPosition_;
121 size_t endIndex = std::min(events_.size(), startIndex + static_cast<size_t>(visibleHeight / lineHeight_) + 1);
122 for (size_t i = startIndex; i < endIndex; ++i) {
123 const GameEvent& event = events_[i];
124 float eventY = y + static_cast<float>(i - startIndex) * lineHeight_ -
125 (static_cast<float>(scrollPosition_) - static_cast<float>(startIndex) * lineHeight_);
126 if (eventY < area_.top + padding_ + 20.0f ||
127 eventY > area_.top + area_.height - padding_) {
128 continue;
129 }
130 std::ostringstream oss;
131 oss << "[" << std::fixed << std::setprecision(1) << event.timestamp << "s] ";
132 sf::Text timestampText;
133 timestampText.setFont(font_);
134 timestampText.setString(oss.str());
135 timestampText.setCharacterSize(10);
136 timestampText.setFillColor(sf::Color(150, 150, 150, 255));
137 timestampText.setPosition(area_.left + padding_, eventY);
138 target.draw(timestampText);
139 sf::Text messageText;
140 messageText.setFont(font_);
141 messageText.setString(event.message);
142 messageText.setCharacterSize(10);
143 messageText.setFillColor(event.color);
144 float timestampWidth = timestampText.getLocalBounds().width;
145 messageText.setPosition(area_.left + padding_ + timestampWidth + 5.0f, eventY);
146 target.draw(messageText);
147 }
148}

150void EventLog::renderScrollbar(sf::RenderTarget& target)
151{
152 if (events_.empty()) return;
153 float contentHeight = static_cast<float>(events_.size()) * lineHeight_;
154 float visibleHeight = area_.height - 2 * padding_ - 20.0f;
155 if (contentHeight <= visibleHeight) return;
156 float thumbHeight = (visibleHeight / contentHeight) * scrollbar_.getSize().y;
157 float thumbPosition = (static_cast<float>(scrollPosition_) / (contentHeight - visibleHeight)) *
158 (scrollbar_.getSize().y - thumbHeight);
159 sf::RectangleShape thumb;
160 thumb.setSize(sf::Vector2f(8.0f, thumbHeight));
161 thumb.setPosition(scrollbar_.getPosition().x,
162 scrollbar_.getPosition().y + thumbPosition);
163 thumb.setFillColor(scrollThumbColor_);
164 target.draw(scrollbar_);
165 target.draw(thumb);
166}

313std::string EventLog::formatTimestamp(float timestamp) const
314{
315 int minutes = static_cast<int>(timestamp) / 60;
316 int seconds = static_cast<int>(timestamp) % 60;
317 int milliseconds = static_cast<int>((timestamp - static_cast<int>(timestamp)) * 1000);
318 return std::to_string(minutes) + ":" +
319 (seconds < 10 ? "0" : "") + std::to_string(seconds) + "." +
320 (milliseconds < 100 ? "0" : "") + (milliseconds < 10 ? "0" : "") + std::to_string(milliseconds);
321}
