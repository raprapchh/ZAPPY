32void Animation::update(float deltaTime)
33{
34 if (!playing_ || frames_.empty()) return;
35 currentTime_ += deltaTime * speed_;
36 if (currentFrame_ < frames_.size()) {
37 float frameDuration = frames_[currentFrame_].duration;
38 if (currentTime_ >= frameDuration) {
39 currentTime_ -= frameDuration;
40 currentFrame_++;
41 if (currentFrame_ >= frames_.size()) {
42 if (loop_) {
43 currentFrame_ = 0;
44 } else {
45 currentFrame_ = frames_.size() - 1;
46 playing_ = false;
47 }
48 }
49 }
50 }
