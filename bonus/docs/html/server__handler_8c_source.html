34static void handle_team_connection(server_t *server, client_t *client,
35 const char *command)
36{
37 team_t *team = team_find_by_name(server->teams, command);
38 char response[64];
39
40 if (team && team_get_slots(team) > 0) {
41 client->type = CLIENT_AI;
42 client->team = team;
43 client->state = STATE_CONNECTED;
44 team_add_client(team);
45 if (setup_ai_client(server, client, team)) {
46 snprintf(response, sizeof(response), "%d\n", team_get_slots(team));
47 client_send(client, response);
48 snprintf(response, sizeof(response), "%d %d\n",
49 server->map->width, server->map->height);
50 client_send(client, response);
51 }
52 } else {
53 client_send(client, "ko\n");

81void server_handle_client_data(server_t *server, client_t *client)
82{
83 char *command;
84
85 if (!client_read(client)) {
86 server_remove_client(server, client);
87 return;
88 }
89 command = client_get_command(client);
90 while (command != NULL) {
91 if (client->state == STATE_WAITING_TEAM &&
92 !process_waiting_team_command(server, client, command)) {
93 free(command);
94 return;
95 }
96 if (client->state == STATE_CONNECTED) {
97 process_connected_client_command(server, client, command);
98 }
99 free(command);
100 command = client_get_command(client);
101 }
102}

116static void process_client_command_queue(server_t *server, client_t *client)
117{
118 command_t *cmd;
119
120 if (client->type != CLIENT_AI || client->state != STATE_CONNECTED)
121 return;
122 if (player_is_frozen(client->player))
123 return;
124 if (client->player->is_busy && get_timestamp() <
125 client->player->busy_until)
126 return;
127 cmd = command_queue_pop(client->cmd_queue);
128 if (cmd) {
129 handle_ai_command(server, client, cmd->data);
130 free(cmd);
131 }
132}

134static void process_events_and_clients(server_t *server, uint64_t now)
135{
136 event_t *event;
137 client_t *client;
138
139 event = event_queue_get_ready(server->events, now);
140 while (event != NULL) {
141 client = find_client_by_fd(server->clients, event->client_fd);
142 if (client && client->state == STATE_CONNECTED &&
143 client->type == CLIENT_AI) {
144 handle_ai_command(server, client, event->command);
145 }
146 event_queue_remove(server->events, event);
147 event = event_queue_get_ready(server->events, now);
148 }
149 client = server->clients;
150 while (client) {
151 process_client_command_queue(server, client);
152 client = client->next;
153 }
154}

156void server_process_events(server_t *server)
157{
158 uint64_t now = get_timestamp();
159 client_t *client;
160
161 process_events_and_clients(server, now);
162 if (now - server->map->last_spawn >=
163 (uint64_t)(20 * 1000 / server->config.freq)) {
164 map_maintain_resources(server->map, server->config.freq);
165 server->map->last_spawn = now;
166 }
167 server_check_players_starvation(server);
168 client = server->clients;
169 while (client) {
170 if (buffer_available(client->out_buf) > 0) {
171 client_write(client);
172 }
173 client = client->next;
174 }
