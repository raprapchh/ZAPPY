20int map_count_resource_type(map_t *map, resource_type_t type)
21{
22 int total = 0;
23 int x = 0;
24 int y = 0;
25
26 if (!map || type < 0 || type >= RESOURCE_TYPES)
27 return 0;
28 for (y = 0; y < map->height; y++) {
29 for (x = 0; x < map->width; x++) {
30 total += map->tiles[y][x].resource_count[type];
31 }
32 }
33 return total;
34}

57static void maintain_resource_type(map_t *map, resource_type_t type)

118static bool remove_resource_from_tile(tile_t *tile, resource_type_t type)
119{
120 resource_t **current;
121 resource_t *to_remove;
122
123 current = &tile->resources;
124 while (*current) {
125 if ((*current)->type == type) {
126 to_remove = *current;
127 *current = (*current)->next;
128 free(to_remove);
129 tile->resource_count[type]--;
130 tile->dirty = true;
131 return true;
132 }
133 current = &(*current)->next;
134 }
135 return false;
136}
