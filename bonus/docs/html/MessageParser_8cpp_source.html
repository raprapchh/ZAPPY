38std::optional<Message> MessageParser::parse(const std::string& rawMessage) const
39{
40 if (rawMessage.empty()) {
41 return std::nullopt;
42 }
43
44 auto tokens = tokenize(rawMessage);
45 if (tokens.empty()) {
46 return std::nullopt;
47 }
48
49 std::string command = tokens[0];
50 std::transform(command.begin(), command.end(), command.begin(), ::tolower);
51
52 auto it = handlers_.find(command);
53 if (it == handlers_.end()) {
54 return Message{MessageType::Unknown, std::move(tokens)};
55 }
56
57 return it->second(tokens);
58}

60std::vector<Message> MessageParser::parseBuffer(std::string& buffer) const
61{
62 std::vector<Message> messages;
63 size_t pos = 0;
64 while ((pos = buffer.find('\n')) != std::string::npos) {
65 std::string line = buffer.substr(0, pos);
66 buffer.erase(0, pos + 1);
67 if (line.empty()) {
68 continue;
69 }
70 if (!line.empty() && line.back() == '\r') {
71 line.pop_back();
72 }
73 auto message = parse(line);
74 if (message) {
75 messages.emplace_back(std::move(*message));
76 }
77 }
78 return messages;
79}

81std::vector<std::string> MessageParser::tokenize(const std::string& message) const
82{
83 std::vector<std::string> tokens;
84 std::istringstream iss(message);
85 std::string token;
86 while (std::getline(iss, token, ' ')) {
87 if (!token.empty()) {
88 tokens.push_back(token);
89 }
90 }
91 return tokens;
