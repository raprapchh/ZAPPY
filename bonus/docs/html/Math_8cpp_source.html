249sf::Color hsvToRgb(float h, float s, float v, sf::Uint8 alpha)
250{
251 h = wrap(h, 0.0f, 1.0f) * 360.0f;
252 s = clamp(s, 0.0f, 1.0f);
253 v = clamp(v, 0.0f, 1.0f);
254 float c = v * s;
255 float x = c * (1.0f - std::abs(std::fmod(h / 60.0f, 2.0f) - 1.0f));
256 float m = v - c;
257 float r, g, b;
258 if (h >= 0 && h < 60) {
259 r = c; g = x; b = 0;
260 } else if (h >= 60 && h < 120) {
261 r = x; g = c; b = 0;
262 } else if (h >= 120 && h < 180) {
263 r = 0; g = c; b = x;
264 } else if (h >= 180 && h < 240) {
265 r = 0; g = x; b = c;
266 } else if (h >= 240 && h < 300) {
267 r = x; g = 0; b = c;
268 } else {
269 r = c; g = 0; b = x;
270 }
271 return sf::Color(
272 static_cast<sf::Uint8>((r + m) * 255),
273 static_cast<sf::Uint8>((g + m) * 255),
274 static_cast<sf::Uint8>((b + m) * 255),
275 alpha
276 );

279void rgbToHsv(const sf::Color& rgb, float& h, float& s, float& v)
280{
281 float r = rgb.r / 255.0f;
282 float g = rgb.g / 255.0f;
283 float b = rgb.b / 255.0f;
284 float maxVal = std::max({r, g, b});
285 float minVal = std::min({r, g, b});
286 float delta = maxVal - minVal;
287 v = maxVal;
288 s = (maxVal == 0.0f) ? 0.0f : delta / maxVal;
289 if (delta == 0.0f) {
290 h = 0.0f;
291 } else if (maxVal == r) {
292 h = 60.0f * (std::fmod((g - b) / delta, 6.0f));
293 } else if (maxVal == g) {
294 h = 60.0f * ((b - r) / delta + 2.0f);
295 } else {
296 h = 60.0f * ((r - g) / delta + 4.0f);
297 }
298 if (h < 0.0f) h += 360.0f;
299 h /= 360.0f;
300}
